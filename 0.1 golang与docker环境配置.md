# goenv
```
mkdir -p /usr/local/src && cd /usr/local/src && wget https://studygolang.com/dl/golang/go1.12.5.linux-amd64.tar.gz
cd /usr/local/src  && tar -xvf go1.12.5.linux-amd64.tar.gz
mkdir -p /home/gopath/{src,pkg,bin}
sed -i '$a\export GOPATH=/home/gopath' /etc/profile
sed -i '$a\export GOROOT=/usr/local/src/go' /etc/profile
sed -i '$a\export GOBIN=$GOROOT/bin'  /etc/profile
sed -i '$a\export PATH=$PATH:$GOPATH:$GOROOT:$GOBIN'  /etc/profile
source /etc/profile
```
